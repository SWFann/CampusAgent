# This directory holds Alembic migration scripts.
# Migration files follow the naming convention:
#   <revision_id>_<short_description>.py
#
# Use the following command to create a new migration:
#   alembic -c apps/api/alembic.ini revision -m "description"
#
# Or with autogenerate (requires a running database):
#   alembic -c apps/api/alembic.ini revision --autogenerate -m "description"
